301 uint8_t getBattlevelDefault(void)
302 {
303 uint16_t voltage = batteryVoltage;
304 uint16_t limit_low = battery_discharge_lim;
305 uint16_t limit_high = battery_charge_lim;
306 uint8_t battLevel;
307
308 if (voltage == 0)
309 {
310 // Unable to measure battery voltage
311 battLevel = 255;
312 }
313 else if (voltage > limit_high)
314 {
315 // External power source
316 battLevel = 0;
317 }
318 else
319 {
320 battLevel = static_cast<uint8_t>(
321 static_cast<float>(voltage - limit_low) / static_cast<float>(limit_high - limit_low) * 255);
322 battLevel = (battLevel == 0) ? 1 : battLevel;
323 battLevel = (battLevel == 255) ? 254 : battLevel;
324 }
325 return battLevel;
326 };

365 uint8_t getBattlevelPowerfeather(void)
366 {
367 Result res;
368 bool supply_good;
369
370 res = Board.checkSupplyGood(supply_good);
371 if ((res == Result::Ok) && supply_good)
372 {
373 return 0; // external power source
374 }
375
376 uint8_t soc = 0;
377 res = Board.getBatteryCharge(soc);
378 log_d("Battery SOC: %u %%", soc);
379 if (res == Result::Ok)
380 {
381 // Scale SOC (0-100%) to LoRaWAN battery level (1-254)
382 uint8_t level;
383 level = static_cast<uint8_t>(static_cast<float>(soc) / 100.0f * 254.0f);
384 level = (level == 0) ? 1 : level;
385 return level;
386 }
387 return 255; // Unable to measure
388 };

502 uint32_t sleepDuration(void)
503 {
504 uint32_t sleep_interval = sleepInterval();
505
506 // If the real time is available, align the wake-up time to
507 // next non-fractional multiple of sleep_interval past the hour
508 if (isRtcSynched())
509 {
510 struct tm timeinfo;
511 time_t t_now = time(nullptr);
512 localtime_r(&t_now, &timeinfo);
513
514 sleep_interval = sleep_interval - ((timeinfo.tm_min * 60) % sleep_interval + timeinfo.tm_sec);
515 }
516
517 sleep_interval = max(sleep_interval, static_cast<uint32_t>(SLEEP_INTERVAL_MIN));
518 return sleep_interval;
519 };

529 void uplinkDelay(uint32_t timeUntilUplink, uint32_t uplinkInterval)
530 {
531 uint32_t uplinkIntervalMs = uplinkInterval * 1000UL;
532 uint32_t delayMs = max(timeUntilUplink, uplinkIntervalMs); // cannot send faster than duty cycle allows
533
534 log_d("Sending uplink in %u s", delayMs / 1000);
535#if defined(ESP32)
536 esp_sleep_enable_timer_wakeup(delayMs * 1000);
537 esp_light_sleep_start();
538#else
539 delay(delayMs);
540#endif
541 };
