205 uint8_t getBattlevel(void)
206 {
207 uint16_t voltage = batteryVoltage;
208 uint16_t limit_low = battery_discharge_lim;
209 uint16_t limit_high = battery_charge_lim;
210 uint8_t battLevel;
211
212 if (voltage == 0)
213 {
214 // Unable to measure battery voltage
215 battLevel = 255;
216 }
217 else if (voltage > limit_high)
218 {
219 // External power source
220 battLevel = 0;
221 }
222 else
223 {
224 battLevel = static_cast<uint8_t>(
225 static_cast<float>(voltage - limit_low) / static_cast<float>(limit_high - limit_low) * 255);
226 battLevel = (battLevel == 0) ? 1 : battLevel;
227 battLevel = (battLevel == 255) ? 254 : battLevel;
228 }
229 return battLevel;
230 };

304 uint32_t sleepDuration(void)
305 {
306 uint32_t sleep_interval = sleepInterval();
307
308 // If the real time is available, align the wake-up time to
309 // next non-fractional multiple of sleep_interval past the hour
310 if (isRtcSynched())
311 {
312 struct tm timeinfo;
313 time_t t_now = time(nullptr);
314 localtime_r(&t_now, &timeinfo);
315
316 sleep_interval = sleep_interval - ((timeinfo.tm_min * 60) % sleep_interval + timeinfo.tm_sec);
317 }
318
319 sleep_interval = max(sleep_interval, static_cast<uint32_t>(SLEEP_INTERVAL_MIN));
320 return sleep_interval;
321 };

331 void uplinkDelay(uint32_t timeUntilUplink, uint32_t uplinkInterval)
332 {
333 uint32_t uplinkIntervalMs = uplinkInterval * 1000UL;
334 uint32_t delayMs = max(timeUntilUplink, uplinkIntervalMs); // cannot send faster than duty cycle allows
335
336 log_d("Sending uplink in %u s", delayMs / 1000);
337#if defined(ESP32)
338 esp_sleep_enable_timer_wakeup(delayMs * 1000);
339 esp_light_sleep_start();
340#else
341 delay(delayMs);
342#endif
343 };
