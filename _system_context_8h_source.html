265 uint8_t getBattlevelDefault(void)
266 {
267 uint16_t voltage = batteryVoltage;
268 uint16_t limit_low = battery_discharge_lim;
269 uint16_t limit_high = battery_charge_lim;
270 uint8_t battLevel;
271
272 if (voltage == 0)
273 {
274 // Unable to measure battery voltage
275 battLevel = 255;
276 }
277 else if (voltage > limit_high)
278 {
279 // External power source
280 battLevel = 0;
281 }
282 else
283 {
284 battLevel = static_cast<uint8_t>(
285 static_cast<float>(voltage - limit_low) / static_cast<float>(limit_high - limit_low) * 255);
286 battLevel = (battLevel == 0) ? 1 : battLevel;
287 battLevel = (battLevel == 255) ? 254 : battLevel;
288 }
289 return battLevel;
290 };

292#if defined(ARDUINO_ESP32S3_POWERFEATHER)
304 uint8_t getBattlevelPowerfeather(void)
305 {
306 Result res;
307 bool supply_good;
308
309 res = Board.checkSupplyGood(supply_good);
310 if ((res == Result::Ok) && supply_good)
311 {
312 return 0; // external power source
313 }
314
315 uint8_t soc = 0;
316 res = Board.getBatteryCharge(soc);
317 log_d("Battery SOC: %u %%", soc);
318 if (res == Result::Ok)
319 {
320 // Scale SOC (0-100%) to LoRaWAN battery level (1-254)
321 uint8_t level;
322 level = static_cast<uint8_t>(static_cast<float>(soc) / 100.0f * 254.0f);
323 level = (level == 0) ? 1 : level;
324 return level;
325 }
326 return 255; // Unable to measure
327 };

401 uint32_t sleepDuration(void)
402 {
403 uint32_t sleep_interval = sleepInterval();
404
405 // If the real time is available, align the wake-up time to
406 // next non-fractional multiple of sleep_interval past the hour
407 if (isRtcSynched())
408 {
409 struct tm timeinfo;
410 time_t t_now = time(nullptr);
411 localtime_r(&t_now, &timeinfo);
412
413 sleep_interval = sleep_interval - ((timeinfo.tm_min * 60) % sleep_interval + timeinfo.tm_sec);
414 }
415
416 sleep_interval = max(sleep_interval, static_cast<uint32_t>(SLEEP_INTERVAL_MIN));
417 return sleep_interval;
418 };

428 void uplinkDelay(uint32_t timeUntilUplink, uint32_t uplinkInterval)
429 {
430 uint32_t uplinkIntervalMs = uplinkInterval * 1000UL;
431 uint32_t delayMs = max(timeUntilUplink, uplinkIntervalMs); // cannot send faster than duty cycle allows
432
433 log_d("Sending uplink in %u s", delayMs / 1000);
434#if defined(ESP32)
435 esp_sleep_enable_timer_wakeup(delayMs * 1000);
436 esp_light_sleep_start();
437#else
438 delay(delayMs);
439#endif
440 };
