212 uint8_t getBattlevel(void)
213 {
214 uint16_t voltage = batteryVoltage;
215 uint16_t limit_low = battery_discharge_lim;
216 uint16_t limit_high = battery_charge_lim;
217 uint8_t battLevel;
218
219 if (voltage == 0)
220 {
221 // Unable to measure battery voltage
222 battLevel = 255;
223 }
224 else if (voltage > limit_high)
225 {
226 // External power source
227 battLevel = 0;
228 }
229 else
230 {
231 battLevel = static_cast<uint8_t>(
232 static_cast<float>(voltage - limit_low) / static_cast<float>(limit_high - limit_low) * 255);
233 battLevel = (battLevel == 0) ? 1 : battLevel;
234 battLevel = (battLevel == 255) ? 254 : battLevel;
235 }
236 return battLevel;
237 };

309 uint32_t sleepDuration(void)
310 {
311 uint32_t sleep_interval = sleepInterval();
312
313 // If the real time is available, align the wake-up time to
314 // next non-fractional multiple of sleep_interval past the hour
315 if (isRtcSynched())
316 {
317 struct tm timeinfo;
318 time_t t_now = time(nullptr);
319 localtime_r(&t_now, &timeinfo);
320
321 sleep_interval = sleep_interval - ((timeinfo.tm_min * 60) % sleep_interval + timeinfo.tm_sec);
322 }
323
324 sleep_interval = max(sleep_interval, static_cast<uint32_t>(SLEEP_INTERVAL_MIN));
325 return sleep_interval;
326 };

336 void uplinkDelay(uint32_t timeUntilUplink, uint32_t uplinkInterval)
337 {
338 uint32_t uplinkIntervalMs = uplinkInterval * 1000UL;
339 uint32_t delayMs = max(timeUntilUplink, uplinkIntervalMs); // cannot send faster than duty cycle allows
340
341 log_d("Sending uplink in %u s", delayMs / 1000);
342#if defined(ESP32)
343 esp_sleep_enable_timer_wakeup(delayMs * 1000);
344 esp_light_sleep_start();
345#else
346 delay(delayMs);
347#endif
348 };
