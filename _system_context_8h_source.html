300 uint8_t getBattlevelDefault(void)
301 {
302 uint16_t voltage = batteryVoltage;
303 uint16_t limit_low = battery_discharge_lim;
304 uint16_t limit_high = battery_charge_lim;
305 uint8_t battLevel;
306
307 if (voltage == 0)
308 {
309 // Unable to measure battery voltage
310 battLevel = 255;
311 }
312 else if (voltage > limit_high)
313 {
314 // External power source
315 battLevel = 0;
316 }
317 else
318 {
319 battLevel = static_cast<uint8_t>(
320 static_cast<float>(voltage - limit_low) / static_cast<float>(limit_high - limit_low) * 255);
321 battLevel = (battLevel == 0) ? 1 : battLevel;
322 battLevel = (battLevel == 255) ? 254 : battLevel;
323 }
324 return battLevel;
325 };

339 uint8_t getBattlevelPowerfeather(void)
340 {
341 Result res;
342 bool supply_good;
343
344 res = Board.checkSupplyGood(supply_good);
345 if ((res == Result::Ok) && supply_good)
346 {
347 return 0; // external power source

349

352 log_d("Battery SOC: %u %%", soc);

354 {
355 // Scale SOC (0-100%) to LoRaWAN battery level (1-254)
356 uint8_t level;
357 level = static_cast<uint8_t>(static_cast<float>(soc) / 100.0f * 254.0f);
358 level = (level == 0) ? 1 : level;
359 return level;
360 }

465 uint32_t sleepDuration(void)
466 {
467 uint32_t sleep_interval = sleepInterval();
468
469 // If the real time is available, align the wake-up time to
470 // next non-fractional multiple of sleep_interval past the hour
471 if (isRtcSynched())
472 {
473 struct tm timeinfo;
474 time_t t_now = time(nullptr);
475 localtime_r(&t_now, &timeinfo);
476
477 sleep_interval = sleep_interval - ((timeinfo.tm_min * 60) % sleep_interval + timeinfo.tm_sec);
478 }
479
480 sleep_interval = max(sleep_interval, static_cast<uint32_t>(SLEEP_INTERVAL_MIN));
481 return sleep_interval;
482 };

492 void uplinkDelay(uint32_t timeUntilUplink, uint32_t uplinkInterval)
493 {
494 uint32_t uplinkIntervalMs = uplinkInterval * 1000UL;
495 uint32_t delayMs = max(timeUntilUplink, uplinkIntervalMs); // cannot send faster than duty cycle allows
496
497 log_d("Sending uplink in %u s", delayMs / 1000);
498#if defined(ESP32)
499 esp_sleep_enable_timer_wakeup(delayMs * 1000);
500 esp_light_sleep_start();
501#else
502 delay(delayMs);
503#endif
504 };
